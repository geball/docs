```graphql Mutation
mutation createThread($input: CreateThreadInput!) {
  createThread(input: $input) {
    thread {
      id
      externalId
      customer {
        id
      }
      status
      statusChangedAt {
        iso8601
        unixTimestamp
      }
      title
      previewText
      priority
      threadFields {
        id
        key
        type
        stringValue
        booleanValue
      }
    }
    error {
      message
      type
      code
      fields {
        field
        message
        type
      }
    }
  }
}
```
